var grahamscanmono = function ( set , i , j , lo ) { var k , n , hi , u , p , q ; n = set.length ; hi = [] ; hi.push( set[i] ) ; hi.push( set[i + 1] ) ; lo.push( set[i] ) ; lo.push( set[i + 1] ) ; p = 0 ; q = 0 ; for ( k = i + 2 ; k < j ; ++k ) { u = set[k] ; while ( p >= 0 && sinsign( hi[p] , hi[p + 1] , u ) >= 0 ) { hi.pop() ; --p ; } hi.push( u ) ; ++p ; while ( q >= 0 && sinsign( lo[q] , lo[q + 1] , u ) <= 0 ) { lo.pop() ; --q ; } lo.push( u ) ; ++q ; }

enumerate hull vertices counter clock wise, in fact ccw if set is monotone increasing, cw otherwise

            * - < - * - < - *
           /                 \

hi[0] = lo[0] hi[p + 1] = lo[q + 1] \ /

            * - > - * - > - *

Note that the first and last elements of hi are droped since they are already in lo
